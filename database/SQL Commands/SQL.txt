CREATE TABLE accounts(
	accountid SERIAL PRIMARY KEY,
	account_type varchar(255),
	firstName varchar(255),
	lastName varchar(255),
	email varchar(255),
	password varchar(255)
);

CREATE TABLE users(
	accountid INT,
	firstName varchar(255),
	lastName varchar(255),
	weight FLOAT,
	age INT,
	feet INT,
	inches int,
	sex varchar(6)
);

CREATE TABLE goals(
	accountid INT,
	steps INT,
	hydration INT,
	calorieIntake INT,
	weight INT,
	sleepHours INT,
	activeMins INT
);

CREATE TABLE dailyLogs(
	accountid INT,
	steps INT,
	hydration INT,
	calorieIntake INT,
	weight INT,
	sleepHours INT,
	activeMins INT
);
