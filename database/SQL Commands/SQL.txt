CREATE TABLE accounts(
	accountid SERIAL PRIMARY KEY,
	account_type varchar(255),
	firstName varchar(255),
	lastName varchar(255),
	email varchar(255),
	password varchar(255)
);

CREATE TABLE users(
	accountid INT,
	firstName varchar(255),
	lastName varchar(255),
	weight FLOAT,
	age INT,
	feet INT,
	inches int,
	sex varchar(6)
);

CREATE TABLE goals(
	accountid INT,
	steps INT,
	hydration INT,
	calorieIntake INT,
	weight INT,
	sleepHours INT,
	activeMins INT
);

CREATE TABLE dailyLogs(
	accountid INT,
	steps INT,
	hydration INT,
	calorieIntake INT,
	weight INT,
	sleepHours INT,
	activeMins INT,
	date DATE
);

CREATE TABLE equipment(
	equipmentid SERIAL PRIMARY KEY,
	purchasedate DATE
);

CREATE TABLE maintenanceLogs(
	loggedby INT,
	equipmentid INT,
	logdate DATE,
	notes varchar(255)
);

CREATE TABLE sessions(
	trainerId INT,
	memberId INT,
	sessionDetails varchar(255),
	sessionType varchar(255),
	startTime varchar(255),
	endTime varchar(255),
	date DATE
);

CREATE TABLE strengthworkshop(
    trainerId INT,
    memberId INT,
    date DATE,
);

CREATE TABLE cardioworkshop(
    trainerId INT,
    memberId INT,
    date DATE,
);

CREATE TABLE yogaworkshop(
    trainerId INT,
    memberId INT,
    date DATE,
);
